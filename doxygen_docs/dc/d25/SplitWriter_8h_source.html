129template<size_t N>
130void SplitWriter<N>::commit()
131{
132 if (_bufNum == INVALID_BUF)
133 {
134 return;
135 }
136 for (size_t i = 0; i < N; i++)
137 {
138 if (i == _bufNum)
139 {
140 // skip the one used as buffer to do it last
141 continue;
142 }
143 auto const dest = _destinations[i];
144 auto const tmp = dest->allocate(_buffer.size());
145 if (tmp.size() < _buffer.size())
146 {
147 drops[i] += 1;
148 continue;
149 }
150 (void)::estd::memory::copy(tmp, _buffer);
151 sent[i] += 1;
152 dest->commit();
153 }
154
155 sent[_bufNum] += 1;
156 _destinations[_bufNum]->commit();
157 _bufNum = INVALID_BUF;
158}
