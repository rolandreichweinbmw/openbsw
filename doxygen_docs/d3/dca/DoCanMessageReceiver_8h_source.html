316template<class DataLinkLayer>
317ReceiveResult DoCanMessageReceiver<DataLinkLayer>::allocated(
318 ::transport::TransportMessage* const message, uint8_t const maxRetryCount)
319{
320 auto const result = DoCanMessageReceiveProtocolHandler<FrameIndexType>::allocated(
321 message != nullptr, maxRetryCount);
322 if ((!DoCanMessageReceiveProtocolHandler<FrameIndexType>::isAllocating())
323 && (message != nullptr))
324 {
325 _message = message;
326 (void)_message->append(_firstFrameData, static_cast<FrameSizeType>(_firstFrameDataSize));
327 }
328 return result;
329}
