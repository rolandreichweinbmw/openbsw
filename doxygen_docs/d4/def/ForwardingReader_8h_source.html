69{
70 _sourceData = _source.peek();

72 // Only forward data, if data is available
73 if (_sourceData.size() > 0)
74 {
75 _destinationData = _destination.allocate(_sourceData.size());
76 if (_destinationData.size() == 0U)
77 {

80 }

84inline void ForwardingReader::release()
85{
86 // If _destinationData is an empty slice, copy will not copy any data.
87 (void)::estd::memory::copy(_destinationData, _sourceData);
88 // Calling commit() will only have an effect if allocate was called before, this is why
89 // we don't need more checks here and can just call commit().
90 _destination.commit();
91 // Invalidate the slices so that calling release() multiple times doesn't copy data by
92 // accident.
93 _destinationData = {};
94 _sourceData = {};
95 _source.release();
