581template<class T, size_t N>
582template<class InputIterator>
583inline typename std::enable_if<std::is_integral<InputIterator>::value == false, void>::type
584vec<T, N>::assign(InputIterator first, InputIterator const last)
585{
586 clear();
587 for (; (!full()) && (first != last); ++first)
588 {
589 // We don't call emplace_back here to avoid the extra estd_assert per call.
590 // We can dereference first because it's an iterator and not necessarily a pointer. Also,
591 // the previous check != last makes it safe to assume that first can be dereferenced.
592 (void)new (&_data[_size]) T(*first);
593 ++_size;
594 }
595}

597template<class T, size_t N>
598inline void vec<T, N>::resize(size_type n)
599{
600 n = estd::min(n, N);
601 for (size_type i = n; i < _size; ++i)
602 {
603 _data[i].cast_to_type()->~T();
604 }
605 for (size_type i = _size; i < n; ++i)
606 {
607 (void)new (&_data[i]) T();
608 }
609 _size = n;
610}
