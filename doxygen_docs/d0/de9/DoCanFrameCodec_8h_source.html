210template<class DataLinkLayer>
211CodecResult DoCanFrameCodec<DataLinkLayer>::decodeFrameType(
212 ::estd::slice<uint8_t const> const& payload, FrameType& frameType) const
213{
214 uint8_t const offset = _config._offset;
215 if (payload.size() <= offset)
216 {
217 return CodecResult::INVALID_FRAME_SIZE;
218 }
219
220 frameType = static_cast<FrameType>((payload[offset] >> 4U) & 0x0FU);
221 return CodecResult::OK;
222}

224template<class DataLinkLayer>
225CodecResult DoCanFrameCodec<DataLinkLayer>::decodeSingleFrame(
226 ::estd::slice<uint8_t const> const& payload,
227 MessageSizeType& messageSize,
228 ::estd::slice<uint8_t const>& data) const
229{
230 size_t const offset = _config._offset;
231 if (checkFrameSize(payload, 2U, _config._singleFrameSize))
232 {
233 messageSize = static_cast<uint8_t>(payload[offset] & 0x0fU);
234 bool const isExtendedSfDlExpected = payload.size() > EXTENDED_SF_DL_EDGE_SIZE;
235 if ((messageSize == 0U) && isExtendedSfDlExpected)
236 {
237 messageSize = payload[offset + 1U];
238 MessageSizeType const checkPayloadSize
239 = messageSize + static_cast<MessageSizeType>(offset) + 2U;
240
241 if ((checkPayloadSize <= static_cast<MessageSizeType>(_config._singleFrameSize._max))
242 && (static_cast<size_t>(checkPayloadSize) <= payload.size()))
243 {
244 data = ::estd::slice<uint8_t const>::from_pointer(
245 &payload[offset + 2U], static_cast<size_t>(messageSize));
246 return CodecResult::OK;
247 }
248 }
249 else if ((messageSize > 0U) && (!isExtendedSfDlExpected))
250 {
251 MessageSizeType const checkPayloadSize
252 = messageSize + static_cast<MessageSizeType>(offset) + 1U;
253 if (checkPayloadSize <= payload.size())
254 {
255 data = ::estd::slice<uint8_t const>::from_pointer(
256 &payload[offset + 1U], static_cast<size_t>(messageSize));
257 return CodecResult::OK;
258 }
259 }
260 else
261 {
262 // invalid message size
263 }
264 return CodecResult::INVALID_MESSAGE_SIZE;
265 }
266 return CodecResult::INVALID_FRAME_SIZE;
267}

269template<class DataLinkLayer>
270CodecResult DoCanFrameCodec<DataLinkLayer>::decodeFirstFrame(
271 ::estd::slice<uint8_t const> const& payload,
272 MessageSizeType& messageSize,
273 FrameIndexType& frameCount,
274 FrameSizeType& consecutiveFrameDataSize,
275 ::estd::slice<uint8_t const>& data) const
276{
277 if (checkFrameSize(payload, 3U, _config._firstFrameSize))
278 {
279 size_t const offset = _config._offset;
280 messageSize
281 = static_cast<MessageSizeType>(::estd::read_be<uint16_t>(&payload[offset]) & 0xFFFU);
282 uint8_t dataStart = 2U;
283 consecutiveFrameDataSize = static_cast<FrameSizeType>(payload.size())
284 - (static_cast<FrameSizeType>(offset) + 1U);
285 if (messageSize == 0U)
286 {
287 if (!checkFrameSize(payload, 6U, _config._firstFrameSize))
288 {
289 return CodecResult::INVALID_FRAME_SIZE;
290 }
291 // Escape sequence, aka message size > 4095 up to uint32_t
292 uint32_t const escapedMessageSize = ::estd::read_be<uint32_t>(&payload[offset + 2U]);
293
294 if ((escapedMessageSize <= ESCAPED_SEQ_MESSAGE_SIZE)
295 || (escapedMessageSize > ::std::numeric_limits<MessageSizeType>::max()))
296 {
297 return CodecResult::INVALID_MESSAGE_SIZE;
298 }
299 messageSize = static_cast<MessageSizeType>(escapedMessageSize);
300 dataStart = 6U;
301 }
302
303 MessageSizeType const count = (messageSize / consecutiveFrameDataSize);
304 if (count >= std::numeric_limits<FrameIndexType>::max())
305 {
306 return CodecResult::INVALID_MESSAGE_SIZE;
307 }
308 frameCount = static_cast<FrameIndexType>(count + 1U);
309 bool const msgFitsIntoLongSingleFrame = fitsIntoLongSingleFrame(messageSize);
310 bool const msgFitsIntoShortSingleFrame = fitsIntoShortSingleFrame(messageSize);
311 if ((frameCount > 1U) && (!msgFitsIntoShortSingleFrame) && (!msgFitsIntoLongSingleFrame))
312 {
313 data = ::estd::slice<uint8_t const>::from_pointer(
314 &payload[offset + dataStart],
315 static_cast<size_t>(consecutiveFrameDataSize) - dataStart + 1U);
316 return CodecResult::OK;
317 }
318 return CodecResult::INVALID_MESSAGE_SIZE;
319 }
320 return CodecResult::INVALID_FRAME_SIZE;
321}

323template<class DataLinkLayer>
324CodecResult DoCanFrameCodec<DataLinkLayer>::decodeConsecutiveFrame(
325 ::estd::slice<uint8_t const> const& payload,
326 uint8_t& sequenceNumber,

328{
329 if (checkFrameSize(payload, 2U, _config._consecutiveFrameSize))
330 {
331 size_t const offset = _config._offset;
332 sequenceNumber = (payload[offset] & 0x0fU);
333 data = ::estd::slice<uint8_t const>::from_pointer(
334 &payload[offset + 1U], payload.size() - (offset + 1U));
335 return CodecResult::OK;
336 }

358template<class DataLinkLayer>
359bool DoCanFrameCodec<DataLinkLayer>::fitsIntoShortSingleFrame(
360 MessageSizeType const messageSize) const
361{
362 FrameSizeType const edge = ::estd::min(EXTENDED_SF_DL_EDGE_SIZE, _config._singleFrameSize._max);
363 uint8_t const offset = _config._offset + 1U;
364 return messageSize <= static_cast<MessageSizeType>(edge) - offset;
365}

367template<class DataLinkLayer>
368bool DoCanFrameCodec<DataLinkLayer>::fitsIntoLongSingleFrame(
369 MessageSizeType const messageSize) const
370{
371 // Ensure _singleFrameSize._max is at least as large as _config._offset + 2
372 estd_assert(_config._singleFrameSize._max >= _config._offset + 2);
373 uint8_t const offset = _config._offset + 2U;
374 return messageSize <= static_cast<MessageSizeType>(_config._singleFrameSize._max) - offset;
375}

377template<class DataLinkLayer>
378CodecResult DoCanFrameCodec<DataLinkLayer>::getEncodedFrameCount(
379 MessageSizeType const messageSize,
380 FrameIndexType& frameCount,
381 FrameSizeType& consecutiveFrameDataSize) const
382{
383 // The max consecutive frame size needs to be at least 2 larger than the configured offset for
384 // this function.
385 estd_assert(_config._consecutiveFrameSize._max >= _config._offset + 2);
386
387 if (messageSize == 0U)
388 {
389 return CodecResult::INVALID_MESSAGE_SIZE;
390 }
391
392 bool const msgFitsIntoLongSingleFrame = fitsIntoLongSingleFrame(messageSize);
393 bool const msgFitsIntoShortSingleFrame = fitsIntoShortSingleFrame(messageSize);
394 if (msgFitsIntoShortSingleFrame || msgFitsIntoLongSingleFrame)
395 {
396 consecutiveFrameDataSize = 0U;
397 frameCount = 1U;
398 }
399 else
400 {
401 consecutiveFrameDataSize = _config._consecutiveFrameSize._max - (_config._offset + 1U);
402 MessageSizeType const count = (messageSize / consecutiveFrameDataSize);
403 if (count >= std::numeric_limits<FrameIndexType>::max())
404 {
405 return CodecResult::INVALID_MESSAGE_SIZE;
406 }
407 frameCount = static_cast<FrameIndexType>(count + 1U);
408 }
409 return CodecResult::OK;
410}

412template<class DataLinkLayer>
413CodecResult DoCanFrameCodec<DataLinkLayer>::encodeDataFrame(
414 ::estd::slice<uint8_t>& payload,
415 ::estd::slice<uint8_t const> const& data,
416 FrameIndexType const frameIndex,
417 FrameSizeType const consecutiveFrameDataSize,
418 FrameSizeType& consumedDataSize) const
419{
420 if ((data.size() == 0U) || (data.size() > ::std::numeric_limits<MessageSizeType>::max()))
421 {
422 return CodecResult::INVALID_FRAME_INDEX;
423 }
424 MessageSizeType const pendingMessageSize = static_cast<MessageSizeType>(data.size());
425
426 FrameSizeType destOffset;
427 FrameSizeType minFrameSize = 0U;
428 if (frameIndex == 0U)
429 {
430 size_t const offset = _config._offset;
431 if (fitsIntoShortSingleFrame(pendingMessageSize))
432 {
433 payload[offset] = static_cast<uint8_t>(FrameType::SINGLE) << 4U;
434 payload[offset] |= static_cast<uint8_t>(pendingMessageSize) & 0x0FU;
435
436 destOffset = static_cast<FrameSizeType>(offset) + 1U;
437 consumedDataSize = static_cast<FrameSizeType>(pendingMessageSize);
438 minFrameSize = _config._singleFrameSize._min;
439 }
440 else if (fitsIntoLongSingleFrame(pendingMessageSize))
441 {
442 payload[offset] = static_cast<uint8_t>(FrameType::SINGLE) << 4U;
443 payload[offset + 1U] = static_cast<uint8_t>(pendingMessageSize);
444 destOffset = static_cast<FrameSizeType>(offset) + 2U;
445 consumedDataSize = static_cast<FrameSizeType>(pendingMessageSize);
446 minFrameSize = _config._singleFrameSize._min;
447 }
448 else if (consecutiveFrameDataSize > 0U)
449 {
450 destOffset = static_cast<FrameSizeType>(offset) + 2U;
451 consumedDataSize = consecutiveFrameDataSize - 1U;
452
453 if (pendingMessageSize <= ESCAPED_SEQ_MESSAGE_SIZE)
454 {
455 ::estd::write_be<uint16_t>(&payload[offset], pendingMessageSize & 0xFFFU);
456 }
457 else
458 {
459 payload[offset] = 0U;
460 payload[offset + 1U] = 0U;
461 ::estd::write_be<uint32_t>(&payload[offset + 2U], pendingMessageSize);
462
463 destOffset += 4U;
464 consumedDataSize -= 4U;
465 }
466 payload[offset] |= static_cast<uint8_t>(FrameType::FIRST) << 4U;
467 minFrameSize = _config._firstFrameSize._min;
468 }
469 else
470 {
471 return CodecResult::INVALID_FRAME_INDEX;
472 }
473 }
474 else if (consecutiveFrameDataSize > 0U)
475 {
476 uint8_t const offset = _config._offset;
477
478 payload[offset] = static_cast<uint8_t>(FrameType::CONSECUTIVE) << 4U;
479 payload[offset] |= static_cast<uint8_t>(frameIndex) & 0x0FU;
480
481 destOffset = offset + 1U;
482 consumedDataSize = static_cast<FrameSizeType>(::estd::min(
483 static_cast<MessageSizeType>(consecutiveFrameDataSize), pendingMessageSize));
484 minFrameSize = ::estd::max(consumedDataSize, _config._consecutiveFrameSize._min);
485 }
486 else
487 {
488 return CodecResult::INVALID_FRAME_INDEX;
489 }
490
491 FrameSizeType const payloadSize = (destOffset + consumedDataSize);
492 if (payloadSize > payload.size())
493 {
494 return CodecResult::INVALID_FRAME_SIZE;
495 }
496
497 (void)::estd::memory::copy(
498 ::estd::slice<uint8_t>::from_pointer(
499 &payload[static_cast<size_t>(destOffset)], static_cast<size_t>(consumedDataSize)),
500 ::estd::slice<uint8_t const>::from_pointer(
501 &data[0U], static_cast<size_t>(consumedDataSize)));
502 return adjustFrame(payload, payloadSize, minFrameSize);
503}

505template<class DataLinkLayer>
506CodecResult DoCanFrameCodec<DataLinkLayer>::encodeFlowControlFrame(
507 ::estd::slice<uint8_t>& payload,
508 FlowStatus const flowStatus,
509 uint8_t const blockSize,
510 uint8_t const encodedMinSeparationTime) const
511{
512 size_t const offset = _config._offset;
513 if (payload.size() < (offset + 3U))
514 {
515 return CodecResult::INVALID_FRAME_SIZE;
516 }
517
518 payload[offset] = static_cast<uint8_t>(FrameType::FLOW_CONTROL) << 4U;
519 payload[offset] |= static_cast<uint8_t>(flowStatus) & 0x0FU;
520
521 payload[offset + 1U] = blockSize;
522 payload[offset + 2U] = encodedMinSeparationTime;
523 return adjustFrame(
524 payload, static_cast<FrameSizeType>(offset) + 3U, _config._flowControlFrameSize._min);
525}

527template<class DataLinkLayer>
528CodecResult DoCanFrameCodec<DataLinkLayer>::adjustFrame(
529 ::estd::slice<uint8_t>& payload,
530 FrameSizeType const payloadSize,
531 FrameSizeType const minFrameSize) const
532{
533 FrameSizeType paddedFrameSize = ::estd::max(payloadSize, minFrameSize);
534 if (!_mapper.mapFrameSize(paddedFrameSize))
535 {
536 return CodecResult::INVALID_FRAME_SIZE;
537 }
538
539 if (paddedFrameSize > payloadSize)
540 {
541 if (paddedFrameSize > payload.size())
542 {
543 return CodecResult::INVALID_FRAME_SIZE;
544 }
545
546 ::estd::memory::set(
547 ::estd::slice<uint8_t>::from_pointer(
548 &payload[static_cast<size_t>(payloadSize)],
549 static_cast<size_t>(paddedFrameSize) - payloadSize),
550 _config._filler);
551 }
552 payload = payload.subslice(static_cast<size_t>(paddedFrameSize));
553 return CodecResult::OK;
554}

556template<class DataLinkLayer>

558 ::estd::slice<uint8_t const> const& payload,
559 FrameSizeType const minPayloadSize,
560 typename DoCanFrameCodecConfig<FrameSizeType>::SizeConfig const frameSize) const
561{
562 return (payload.size() >= static_cast<size_t>(_config._offset) + minPayloadSize)

564 && (payload.size() <= static_cast<size_t>(frameSize._max));
