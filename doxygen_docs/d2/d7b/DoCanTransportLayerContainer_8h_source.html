179template<class DataLinkLayer>
180void DoCanTransportLayerContainer<DataLinkLayer>::shutdown(
181 ShutdownCallbackType const shutdownCallback)
182{
183 _shutdownPendingCount = 1U;
184 _shutdownCallback = shutdownCallback;
185 for (TransportLayerType& layer : _layers)
186 {
187 {
188 // Even if the value changes between now and the increment, we want to ensure this
189 // function is never about to be incremented when _shutdownPendingCount is ever the max
190 // possible value.
191 estd_assert(
192 _shutdownPendingCount
193 != std::numeric_limits<decltype(_shutdownPendingCount)>::max());
194 ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
195 ++_shutdownPendingCount;
196 }
197 (void)layer.shutdown(ShutdownDelegate::create<
198 DoCanTransportLayerContainer<DataLinkLayer>,
199 &DoCanTransportLayerContainer<DataLinkLayer>::shutdownDone>(*this));
200 }
201 singleShutdownDone();
202}

234template<class DataLinkLayer>
235void DoCanTransportLayerContainer<DataLinkLayer>::singleShutdownDone()
236{
237 bool done;
238 {
239 // Even if the value changes between now and the decrement, we want to ensure this function
240 // is never called when _shutdownPendingCount is ever 0.
241 estd_assert(_shutdownPendingCount != 0);
242 ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
243 --_shutdownPendingCount;
244 done = (_shutdownPendingCount == 0U);
245 }
246 if (done)
247 {
248 _shutdownCallback();
249 }
250}
