140struct string_view_tokens
141{
142 string_view_tokens() = default;
143
144 string_view_tokens(string_view const& source, string_view const& delims) : _delims(delims)
145 {
146 auto first_token = source;
147 first_token.skip_while_matches(_delims);
148
149 _current_token = first_token;
150 _next_token = first_token;
151 // advance into first token
152 this->operator++();
153 }
154
155 string_view_tokens& operator++()
156 {
157 // Advance current token to next token.
158 _current_token = _next_token;
159 // Find next delimiter
160 _next_token.skip_until_matches(_delims);
161 // trim current token up to next delimiter
162 _current_token.remove_suffix(_next_token.length());
163 // find start of next token
164 _next_token.skip_while_matches(_delims);
165
166 return *this;
167 }
168
169 bool operator==(string_view_tokens const& other) const
170 {
171 return (_current_token == other._current_token) && (_next_token == other._next_token);
172 }
173
174 bool operator!=(string_view_tokens const& other) const { return !(*this == other); }
175
176 string_view const operator*() const { return _current_token; }
177
178 string_view_tokens& begin() { return *this; }
179
180 static string_view_tokens end() { return string_view_tokens(); }
181
182private:
183 string_view _current_token;
184 string_view _next_token;
185 // string_view has a slice which uses NSDMI; therefore we shouldn't need it here, but clang
186 // deletes default constructor (-Wdefaulted-function-deleted) when it shouldn't
187 string_view const _delims{};
188};

194}
