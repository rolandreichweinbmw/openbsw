182 /* Implementation of SWAR bitcount algorithm based on:
183 https://graphics.stanford.edu/~seander/bithacks.html#CountBitsSetParallel */
184 size_t count() const
185 {
186 constexpr value_type m1 = std::numeric_limits<value_type>::max() / 3U; // binary: 0101...
187 constexpr value_type m2
188 = std::numeric_limits<value_type>::max() / 15U * 3U; // binary: 00110011..
189 constexpr value_type m4
190 = std::numeric_limits<value_type>::max() / 255U * 15U; // binary: 00001111 ...
191 constexpr value_type h01 = std::numeric_limits<value_type>::max()
192 / 255U; // the sum of 256 to the power of 0,1,2,3...
193 value_type tmp = _value;
194 tmp = tmp - ((tmp >> 1U) & m1);
195 tmp = (tmp & m2) + ((tmp >> 2U) & m2);
196 tmp = (tmp + (tmp >> 4U)) & m4;
197 tmp = static_cast<value_type>(tmp * h01) >> ((sizeof(value_type) - 1U) * 8U);
198 return tmp;
199 }

532 void operator>>=(size_t const pos)
533 {
534 size_t const shift = pos % (sizeof(value_type) * 8);
535 size_t const zeros = pos / (sizeof(value_type) * 8);
536
537 for (size_t i = 0; i < _data.size(); ++i)
538 {
539 if (i < _data.size() - zeros - 1)
540 {
541 _data[i] = _data[i + zeros];
542 _data[i] >>= shift;
543 // In case shift == 0, just return 0 to prevent a shift of 32 bits
544 // (technically undefined behavior)
545 auto const left
546 = (shift == 0) ? 0 : _data[i + 1 + zeros] << (sizeof(value_type) * 8 - shift);
547 _data[i] = left | _data[i];
548 }
549 else if (i == _data.size() - zeros - 1)
550 {
551 _data[i] >>= shift;
552 }
553 else
554 {

556 }
557 }

563 size_t const zeros = pos / (sizeof(value_type) * 8);
564
565 for (size_t i = _data.size(); i > 0; --i)
566 {
567 auto const idx = i - 1;
568 if (idx > zeros)
569 {
570 _data[idx] = _data[idx - zeros];
571 _data[idx] <<= shift;
572 // In case shift == 0, just return 0 to prevent a shift of 32 bits (technically
573 // undefined behavior)
574 auto const right
575 = (shift == 0) ? 0
576 : (_data[idx - 1 - zeros] >> (sizeof(value_type) * 8 - shift));
577 _data[idx] = right | _data[idx];
578 }
579 else if (idx == zeros)
580 {
581 _data[idx] <<= shift;
582 }
583 else
584 {
585 _data[idx] = 0;
586 }
587 }
