296template<class DataLinkLayer>
297TransmitResult DoCanMessageTransmitter<DataLinkLayer>::handleFlowControl(
298 FlowStatus const flowStatus,
299 uint8_t const blockSize,
300 uint32_t const minSeparationTimeUs,
301 uint8_t const maxWaitCount)
302{
303 TransmitResult const result
304 = DoCanMessageTransmitProtocolHandler<FrameIndexType>::handleFlowControl(

306 if (result.getActionSet().test(TransmitAction::STORE_SEPARATION_TIME))

309 }
