168class DiagnosisConfiguration : public AbstractDiagnosisConfiguration

170public:
174 explicit DiagnosisConfiguration(
175 uint16_t const diagAddress,
176 uint16_t const broadcastAddress,
177 uint8_t const busId,
178 uint16_t const maxResponsePayloadSize,
179 bool const activateOutgoingPending,
180 bool const acceptAllRequests,
181 bool const copyFunctionalRequests,
182 ::async::ContextType const context)
183 : AbstractDiagnosisConfiguration(
184 diagAddress,
185 broadcastAddress,
186 busId,
187 maxResponsePayloadSize,
188 activateOutgoingPending,
189 acceptAllRequests,
190 copyFunctionalRequests,
191 fIncomingDiagConnectionPool,
192 fSendJobQueue,
193 context)
194 , fIncomingDiagConnectionPool()
195 , fOutgoingDiagConnections()
196 , fSendJobQueue()
197 , fResponseQueues()
198 {
199 // Because we must not use fResponseQueues polymorphical, i.e. passing
200 // it as an array of ::estd::queue to AbstractDiagnosisConfiguration
201 // we need to convert it first
202 ::estd::queue<transport::TransportJob>* baseQueues[NUM_OUTGOING_CONNECTIONS];
203 for (uint8_t i = 0U; i < NUM_OUTGOING_CONNECTIONS; ++i)
204 {
205 baseQueues[i] = &fResponseQueues[i];
206 }
207 AbstractDiagnosisConfiguration::init(
208 fOutgoingDiagConnections, baseQueues, NUM_OUTGOING_CONNECTIONS);
209 }
210
211private:
212 ::estd::declare::object_pool<IncomingDiagConnection, NUM_INCOMING_CONNECTIONS>
213 fIncomingDiagConnectionPool;
214 ManagedOutgoingDiagConnection fOutgoingDiagConnections[NUM_OUTGOING_CONNECTIONS];
215 ::estd::declare::queue<transport::TransportJob, MAX_NUM_INCOMING_MESSAGES> fSendJobQueue;
216 ::estd::declare::queue<transport::TransportJob, MAX_NUM_INCOMING_MESSAGES>
217 fResponseQueues[NUM_OUTGOING_CONNECTIONS];
218};
