41template<uint8_t IndexUpperBound, class Crc8>
42void PersistentComponentConfig<IndexUpperBound, Crc8>::readLevels()
43{
44 ::estd::slice<uint8_t const> const readBuffer = _persistenceManager.readMapping(_buffer);
45 if (readBuffer.size() > 1U)
46 {
47 uint8_t componentCount = static_cast<uint8_t>(readBuffer.size()) - 1U;
48 if (componentCount > IndexUpperBound)
49 {
50 componentCount = IndexUpperBound;
51 }
52 uint8_t const crc = readBuffer[0U];
53 if (crc == getComponentCrc(componentCount))
54 {
55 for (uint8_t idx = 0U; idx < componentCount; ++idx)
56 {
57 this->setLevel(
58 idx,
59 static_cast<::util::logger::Level>(readBuffer[static_cast<size_t>(idx) + 1U]));
60 }
61 }
62 }
63}

67{
