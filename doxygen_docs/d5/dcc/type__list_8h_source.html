63template<typename T, typename TAIL = type_list_end>
64struct type_list
65{
66 using type = T;
67 using tail = TAIL;
68
69 static constexpr size_t size() { return 1 + tail::size(); }
70
71 template<typename U, size_t INDEX = 0U>
72 static constexpr size_t index_of()
73 {
74 return (std::is_same<T, U>::value) ? INDEX : tail::template index_of<U, INDEX + 1>();
75 }
76
77 template<typename U>
78 static constexpr bool contains()
79 {
80 return (std::is_same<T, U>::value) ? true : tail::template contains<U>();
81 }
82
83 static constexpr size_t max_size()
84 {
85 return ::estd::max<size_t>(size_of<type>::value, tail::max_size());
86 }
87
88 static constexpr size_t max_align()
89 {
90 return ::estd::max<size_t>(align_of<type>::value, tail::max_align());
91 }
92
93 template<template<class> class Trait>
94 static constexpr bool all_types()
95 {
96 return ((Trait<type>::value) && (tail::template all_types<Trait>()));
97 }
98};
