693template<typename ToDur, typename Rep, typename Period>
694ToDur duration_cast(duration<Rep, Period> const& d)
695{
696 using todur_t = typename enable_if<internal::_is_duration<ToDur>::value, ToDur>::type;
697 using to_period_t = typename ToDur::period;
698 using to_rep_t = typename ToDur::rep;
699 using cf_t = ::estd::ratio_divide<Period, to_period_t>;
700 using cr_t = typename ::estd::internal::_common_integral_type<
701 typename ::estd::internal::_common_integral_type<to_rep_t, Rep>::type,
702 intmax_t>::type;
703 using dc_t = internal::_duration_cast_impl<todur_t, cf_t, cr_t>;
704 return dc_t::_cast(d);
705}
