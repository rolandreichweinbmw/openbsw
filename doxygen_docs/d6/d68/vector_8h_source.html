703template<class T>
704vector<T>& vector<T>::operator=(vector<T> const& other)
705{
706 // identical vector
707 if (this == &other)
708 {
709 return *this;
710 }
711
712 estd_assert(_max_size >= other._size);
713
714 if (size() >= other.size())
715 {
716 (void)std::copy(other.cbegin(), other.cend(), begin());
717 (void)erase(begin() + other.size(), end());
718 }
719 else
720 {
721 (void)std::copy(other.cbegin(), other.cbegin() + size(), begin());
722 (void)insert(begin() + size(), other.cbegin() + size(), other.cend());
723 }
724
725 return *this;
726}

962template<class T>
963inline void
964vector<T>::insert(const_iterator const position, size_type const n, const_reference value)
965{
966 estd_assert((size() + n) <= max_size());
967 iterator insertItr = begin() + (position - cbegin());
968 auto const diffToEnd = static_cast<difference_type>(cend() - position);
969 auto const numToMove = static_cast<size_type>(diffToEnd);
970
971 if (n > numToMove) // appended elements will consist of moved ones and new copies of 'value'
972 {
973 // append new copies of 'value'
974 for (size_t i = 0; i < n - numToMove; ++i)
975 {
976 (void)emplace_back().construct(value);
977 }
978 // append new elements that are copies of existing ones
979 iterator srcItr = insertItr;
980 for (size_t i = n - numToMove; i < n; ++i)
981 {
982 (void)emplace_back().construct(::std::move(*srcItr));
983 ++srcItr;
984 }
985 // create remaining new copies of 'value' at the given position
986 for (size_t i = 0; i < numToMove; ++i)
987 {
988 *insertItr = value;
989 ++insertItr;
990 }
991 }
992 else // appended elements will consist of moved ones only
993 {
994 // append new elements that are copies of existing elements
995 iterator srcItr = end() - n;
996 for (size_t i = 0; i < n; ++i)
997 {
998 (void)emplace_back().construct(::std::move(*srcItr));
999 ++srcItr;
1000 }
1001 // make more room by moving existing elements in reverse order
1002 srcItr = end() - 2U * n - 1;
1003 iterator dstItr = end() - n - 1;
1004 for (size_t i = 0; i < numToMove - n; ++i)
1005 {
1006 *dstItr = ::std::move(*srcItr);
1007 --dstItr;
1008 --srcItr;
1009 }
1010 // create new copies of 'value' at the given position
1011 for (size_t i = 0; i < n; ++i)
1012 {
1013 *insertItr = value;
1014 ++insertItr;
1015 }
1016 }

1031template<class T>
1032typename vector<T>::iterator vector<T>::erase(const_iterator const first, const_iterator const last)
1033{
1034 estd_assert(last <= cend());
1035 if (last < first) // should this be an assert?
1036 {
1037 return begin() + (last - cbegin());
1038 }
1039 // move all elements to front
1040 iterator const srcIter = begin() + (last - cbegin());
1041 iterator dstItr = begin() + (first - cbegin());
1042 iterator const returnItem = dstItr;
1043 dstItr = ::std::move(srcIter, end(), dstItr);
1044
1045 // destroy empty (after move) elements
1046 while (dstItr != end())
1047 {
1048 (*dstItr).~T();
1049 ++dstItr;
1050 }
1051
1052 difference_type const numErased = last - first;
1053 _size -= static_cast<size_type>(numErased);
1054 return returnItem;
1055}
