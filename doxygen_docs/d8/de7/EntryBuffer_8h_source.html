70template<uint8_t MaxBufferSize, class E>
71void EntryBuffer<MaxBufferSize, E>::addEntry(::estd::slice<uint8_t const> const& entryBuffer)
72{
73 size_t entrySize = entryBuffer.size();
74 uint8_t const* src = entryBuffer.data();
75 if (entrySize >= MaxBufferSize)
76 {
77 entrySize = MaxBufferSize;
78 }
79 size_t const neededSize = entrySize + 1U;
80
81 while ((_usedSize + neededSize) > _size)
82 {
83 ++_firstEntryIndex;
84 size_t const removeBufferSize = static_cast<size_t>(*_firstEntry) + 1U;
85 movePointer(_firstEntry, removeBufferSize);
86 _usedSize -= removeBufferSize;
87 }
88 *_write = static_cast<uint8_t>(entrySize);
89 movePointer(_write, 1U);
90
91 if ((_write + entrySize) > _bufferEnd)
92 {
93 ptrdiff_t const partSize = _bufferEnd - _write;
94 (void)memcpy(_write, src, static_cast<size_t>(partSize));
95 src += partSize;
96 _write = _bufferStart;
97 entrySize -= partSize;
98 }
99 (void)memcpy(_write, src, entrySize);
100 movePointer(_write, entrySize);
101 _usedSize += neededSize;
102 ++_writeEntryIndex;
103}

105template<uint8_t MaxBufferSize, class E>
106uint8_t EntryBuffer<MaxBufferSize, E>::getNextEntry(
107 ::estd::slice<uint8_t> const& entryBuffer, EntryRef& entryRef) const
108{
109 uint8_t const* readPos = entryRef.getReadPointer();
110 E entryIndex = entryRef.getIndex();
111
112 SignedIndexType const entryDiff = static_cast<SignedIndexType>(entryIndex - _firstEntryIndex);
113 if ((readPos == nullptr) || (entryDiff < 0))
114 {
115 entryIndex = _firstEntryIndex;
116 readPos = _firstEntry;
117 }
118 if (entryIndex == _writeEntryIndex)
119 {
120 return 0;
121 }
122
123 uint8_t entrySize = *readPos;
124 movePointer(readPos, 1U);
125 uint8_t const* readUpdate = readPos;
126 movePointer(readUpdate, static_cast<size_t>(entrySize));
127 if (entrySize > entryBuffer.size())
128 {
129 entrySize = static_cast<uint8_t>(entryBuffer.size());
130 }
131 uint8_t copySize = entrySize;
132 uint8_t* dest = entryBuffer.data();
133
134 if ((readPos + copySize) > _bufferEnd)
135 {
136 ptrdiff_t const partSize = _bufferEnd - readPos;
137 (void)memcpy(dest, readPos, static_cast<size_t>(partSize));
138 dest += partSize;
139 readPos = _bufferStart;
140 copySize -= static_cast<uint8_t>(partSize);
141 }
142 (void)memcpy(dest, readPos, static_cast<size_t>(copySize));
143 entryRef.update(entryIndex + 1U, readUpdate);
144 return entrySize;
145}
