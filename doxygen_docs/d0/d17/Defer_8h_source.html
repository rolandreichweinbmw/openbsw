14// C++11 version of std::exchange for internal use.
15template<typename T, typename U = T>
16inline T exchange(T& obj, U&& new_val)
17{
18 T const old_val = std::move(obj);
19 obj = std::forward<U>(new_val);
20 return old_val;
21}
22} // namespace internal

27template<class F>
28class Defer
29{
30public:
31 static_assert(
32 (!std::is_reference<F>::value) && (!std::is_const<F>::value)
33 && (!std::is_volatile<F>::value),
34 "Defer should store its callable by value");
35
36 Defer(Defer const&) = delete;
37 Defer& operator=(Defer const&) = delete;
38 Defer& operator=(Defer&&) = delete;
39
40 explicit Defer(F f) : _f{std::move(f)}, _invoke{true} {}
41
42 Defer(Defer&& other)
43 : _f{std::move(other._f)}, _invoke{internal::exchange(other._invoke, false)}
44 {}
45
46 ~Defer()
47 {
48 if (_invoke)
49 {
50 _f();
51 }
52 }
53
54private:
55 F _f;
56 bool _invoke;
57};
