46template<class Codec>
47CodecResult DoCanFrameDecoder<Codec>::decodeFrame(
48 DoCanConnection<DataLinkLayerType> const& connection,
49 ::estd::slice<uint8_t const> const& payload,
50 IDoCanFrameReceiver<DataLinkLayerType>& receiver)
51{
52 CodecType const& codec = connection.getFrameCodec();
53 FrameType frameType;
54 CodecResult result = codec.decodeFrameType(payload, frameType);
55 if (result == CodecResult::OK)
56 {
57 switch (frameType)
58 {
59 case FrameType::SINGLE:
60 {
61 ::estd::slice<uint8_t const> data;
62 MessageSizeType messageSize;
63 result = codec.decodeSingleFrame(payload, messageSize, data);
64 if (result == CodecResult::OK)
65 {
66 receiver.firstDataFrameReceived(connection, messageSize, 1U, 0U, data);
67 }
68 break;
69 }
70 case FrameType::FIRST:
71 {
72 ::estd::slice<uint8_t const> data;
73 MessageSizeType messageSize;
74 FrameIndexType frameCount;
75 FrameSizeType consecutiveDataFrameSize;
76 result = codec.decodeFirstFrame(
77 payload, messageSize, frameCount, consecutiveDataFrameSize, data);
78 if (result == CodecResult::OK)
79 {
80 receiver.firstDataFrameReceived(
81 connection, messageSize, frameCount, consecutiveDataFrameSize, data);
82 }
83 break;
84 }
85 case FrameType::CONSECUTIVE:
86 {
87 ::estd::slice<uint8_t const> data;
88 uint8_t sequenceNumber;
89 result = codec.decodeConsecutiveFrame(payload, sequenceNumber, data);
90 if (result == CodecResult::OK)
91 {
92 receiver.consecutiveDataFrameReceived(
93 connection.getDataLinkAddressPair().getReceptionAddress(),
94 sequenceNumber,
95 data);
96 }
97 break;
98 }
99 case FrameType::FLOW_CONTROL:
100 {
101 FlowStatus flowStatus;
102 uint8_t blockSize;
103 uint8_t encodedMinSeparationTime;
104 result = codec.decodeFlowControlFrame(
105 payload, flowStatus, blockSize, encodedMinSeparationTime);
106 if (result == CodecResult::OK)
107 {
108 receiver.flowControlFrameReceived(
109 connection.getDataLinkAddressPair().getReceptionAddress(),
110 flowStatus,
111 blockSize,
112 encodedMinSeparationTime);
113 }
114 break;
115 }
116 default:
117 {
118 return CodecResult::INVALID_FRAME_TYPE;
119 }
120 }
121 }
122 return result;
123}
