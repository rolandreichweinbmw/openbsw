884template<class T>
885void forward_list<T>::pop_back()
886{
887 if (empty())
888 {
889 return;
890 }
891 auto previous = before_begin();
892 auto current = before_begin();
893 auto peek = begin();
894 auto const endIter = end();
895 while (peek != endIter)
896 {
897 previous = current;
898 current = peek;
899 ++peek;
900 }
901 current->node_::set_free();
902 previous._current->set_next(nullptr);
903}

951template<class T>
952inline void forward_list<T>::remove(forward_list_node<T>& node)
953{
954 if (empty())
955 {
956 return;
957 }
958
959 auto current = before_begin();
960 auto previous = end();
961 auto const endIter = end();
962 while ((current != endIter) && (current._current != &node))
963 {
964 previous = current;
965 ++current;
966 }
967
968 if (current == endIter)
969 { // node isn't in this list!
970 return;
971 }
972
973 previous._current->set_next(node.next());
974
975 node.set_free();
976}

998void forward_list<T>::remove_if(Predicate pred, Removed& removed)
999{
1000 forward_list tmp;
1001 iterator inItr = tmp.before_begin();
1002 while (!empty())
1003 {
1004 reference current = front();
1005 pop_front();
1006 if (pred(current) == false)
1007 {
1008 (void)tmp.insert_after(inItr, current);
1009 ++inItr;
1010 }
1011 else
1012 {
1013 removed.push_front(current);

1098void forward_list<T>::splice_after(
1099 iterator const position, iterator const first, iterator const last)
1100{
1101 if (position != first)
1102 {
1103 iterator element = ::estd::next(first);
1104 if ((position != element) && (last != element))
1105 {
1106 // first points to last after this
1107 first._current->set_next(last._current);
1108
1109 node_* const tmp = position._current->next();
1110 position._current->set_next(element._current);
1111 while (element._current->next() != last._current)
1112 {
1113 ++element;
1114 }
1115 element._current->set_next(tmp);
1116 }
1117 }

1138{

1148 if (next == nullptr)

1183template<class T>
1184template<class Compare>
1185void forward_list<T>::merge(forward_list& other, Compare const compare)
1186{
1187 auto current = before_begin();
1188 auto next = ::estd::next(current);
1189 auto const endIter = end();
1190
1191 // iterate until end of this list or until other is empty
1192 while ((next != endIter) && (!other.empty()))
1193 {
1194 // check if first element of other goes before current->next
1195 if (compare(*other.begin(), *next) == true)
1196 {
1197 node_* const other_before_first = &other._before_first;
1198 node_* const other_first = other_before_first->next();
1199 // remove first element in other
1200 other_before_first->set_next(other_first->next());
1201 // move other's former first before current
1202 other_first->set_next(current._current->next());
1203 current._current->set_next(other_first);
1204 }
1205 ++current;
1206 next = ::estd::next(current);
1207 }
1208 // append rest of other if there is any
1209 if (!other.empty())
1210 {
1211 current._current->set_next(other._before_first.next());
1212 other._before_first.set_next(nullptr);
1213 }
1214}

1230 auto list = begin();
1231 auto const endIter = end();
1232
1233 size_type list_size = 1U;
1234 size_type num_merges = 0U;
1235
1236 do
1237 {
1238 num_merges = 0U;
1239 iterator left = list;
1240 iterator tail = endIter;
1241 list = endIter;
1242 iterator next;
1243 while (left != endIter)
1244 {
1245 ++num_merges;
1246 iterator right = left;
1247 size_t left_size = 0U;
1248 size_t right_size = list_size;
1249 // cut lists into halves
1250 while ((right != endIter) && (left_size < list_size))
1251 {
1252 ++left_size;
1253 ++right;
1254 }
1255 // run through lists and append
1256 while ((left_size > 0U) || ((right_size > 0U) && (right != endIter)))
1257 {
1258 if (left_size == 0U)
1259 { // left is empty -> take right
1260 next = right;
1261 ++right;
1262 --right_size;
1263 }
1264 else if ((right_size == 0U) || (right == endIter))
1265 { // right is empty -> take left
1266 next = left;
1267 ++left;
1268 --left_size;
1269 }
1270 else if (compare(*right, *left) == true)
1271 { // right goes before left
1272 next = right;
1273 ++right;
1274 --right_size;
1275 }
1276 else
1277 { // left goes before right
1278 next = left;
1279 ++left;
1280 --left_size;
1281 }
1282 if (tail != endIter)
1283 {
1284 tail->node_::set_next(next._current);
1285 }
1286 else
1287 {
1288 list = next;
1289 }
1290 tail = next;
1291 }
1292 // right is now after the just sorted part, start next round here
1293 left = right;
1294 }
1295 // terminate tail
1296 tail->node_::set_next(nullptr);
1297 // double list size
1298 list_size <<= 1U;
1299 } while (num_merges > 1U);
1300 _before_first.set_next(list._current);
1301}
