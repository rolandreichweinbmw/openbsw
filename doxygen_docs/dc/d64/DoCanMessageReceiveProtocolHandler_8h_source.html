276template<typename FrameIndexType>
277inline ReceiveResult DoCanMessageReceiveProtocolHandler<FrameIndexType>::allocated(
278 bool const success, uint8_t const maxRetryCount)
279{
280 if (!_isAllocating)
281 {
282 return ReceiveResult(true).setMessage(
283 ReceiveMessage::ILLEGAL_STATE, static_cast<uint8_t>(_state));
284 }
285
286 if (success)
287 {
288 _isAllocating = false;
289 _allocateRetryCount = 0U;
290 return setState((_frameCount == 1U) ? ReceiveState::PROCESSING : ReceiveState::SEND);
291 }
292
293 if (_frameCount == 1U)
294 {
295 ++_allocateRetryCount;
296 if (_allocateRetryCount > 1U)
297 {
298 return setDone(ReceiveMessage::ALLOCATION_RETRY_COUNT_EXCEEDED);
299 }
300
301 return setState(ReceiveState::WAIT, ReceiveTimeout::ALLOCATE);
302 }
303
304 ++_allocateRetryCount;
305 if (_allocateRetryCount > maxRetryCount)
306 {
307 return setDone(ReceiveMessage::ALLOCATION_RETRY_COUNT_EXCEEDED);
308 }
309
310 return setState(ReceiveState::SEND);
311}

331template<typename FrameIndexType>
332inline ReceiveResult DoCanMessageReceiveProtocolHandler<FrameIndexType>::consecutiveFrameReceived(
333 uint8_t const frameIndex, uint8_t const maxBlockSize)
334{
335 if ((_state != ReceiveState::WAIT) || (_timeout != ReceiveTimeout::RX))
336 {
337 return ReceiveResult(false);
338 }
339
340 if ((_frameIndex & 0xFU) != frameIndex)
341 {
342 return setDone(ReceiveMessage::BAD_SEQUENCE_NUMBER, frameIndex);
343 }
344
345 ++_frameIndex;
346 if (_frameIndex == _frameCount)
347 {
348 return setState(ReceiveState::PROCESSING);
349 }
350 if (maxBlockSize > 0U)
351 {
352 ++_blockFrameIndex;
353 if (_blockFrameIndex == maxBlockSize)
354 {
355 _blockFrameIndex = 0U;
356 return setState(ReceiveState::SEND);
357 }
358 }
359 // set state to wait for timeout as in the default case
360 return setState(ReceiveState::WAIT, ReceiveTimeout::RX);
361}
