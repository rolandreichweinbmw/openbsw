117template<uintmax_t N1, uintmax_t N0, uintmax_t D>
118struct _big_div_impl
119{
120private:
121 static constexpr uintmax_t _c
122 = static_cast<uintmax_t>(1U) << static_cast<uintmax_t>(sizeof(intmax_t) * 4U);
123 static constexpr uintmax_t _d1 = D / _c;
124 static constexpr uintmax_t _d0 = D % _c;
125
126 static constexpr uintmax_t _q1x = static_cast<uintmax_t>(N1 / _d1);
127 static constexpr uintmax_t _r1x = static_cast<uintmax_t>(N1 % _d1);
128 static constexpr uintmax_t _m = static_cast<uintmax_t>(_q1x * _d0);
129 static constexpr uintmax_t _r1y = (_r1x * _c) + (N0 / _c);
130 static constexpr uintmax_t _r1z = _r1y + D;
131 static constexpr uintmax_t _r1 = static_cast<uintmax_t>(
132 ((_r1y < _m) ? ((_r1z >= D) && (_r1z < _m)) ? (_r1z + D) : _r1z : _r1y) - _m);
133 static constexpr uintmax_t _q1 = static_cast<uintmax_t>(
134 _q1x - ((_r1y < _m) ? ((_r1z >= D) && (_r1z < _m)) ? static_cast<uintmax_t>(2U) : 1U : 0U));
135 static constexpr uintmax_t _q0x = static_cast<uintmax_t>(_r1 / _d1);
136 static constexpr uintmax_t _r0x = static_cast<uintmax_t>(_r1 % _d1);
137 static constexpr uintmax_t _n = static_cast<uintmax_t>(_q0x * _d0);
138 static constexpr uintmax_t _r0y = (_r0x * _c) + (N0 % _c);
139 static constexpr uintmax_t _r0z = _r0y + D;
140 static constexpr uintmax_t _r0 = static_cast<uintmax_t>(
141 ((_r0y < _n) ? ((_r0z >= D) && (_r0z < _n)) ? (_r0z + D) : _r0z : _r0y) - _n);
142 static constexpr uintmax_t _q0 = static_cast<uintmax_t>(
143 _q0x - ((_r0y < _n) ? (((_r0z >= D) && (_r0z < _n)) ? 2U : 1U) : 0U));

145public:
146 static constexpr uintmax_t _quot = static_cast<uintmax_t>((_q1 * _c) + _q0);
147 static constexpr uintmax_t _rem = _r0;

149private:
150 using prod_t = _big_mul<_quot, D>;
151 using sum_t = _big_add<prod_t::_hi, prod_t::_lo, 0, _rem>;

154template<uintmax_t N1, uintmax_t N0, uintmax_t D>
155struct _big_div
156{
157private:
158 static constexpr uintmax_t _shift = _static_clz<D>::value;
159 static constexpr uintmax_t _coshift_
160 = static_cast<uintmax_t>((sizeof(uintmax_t) * static_cast<uintmax_t>(8U)) - _shift);
161 static constexpr uintmax_t _coshift = static_cast<uintmax_t>((_shift != 0U) ? _coshift_ : 0U);
162 static constexpr uintmax_t _c1 = static_cast<uintmax_t>(static_cast<uintmax_t>(1U) << _shift);
163 static constexpr uintmax_t _c2 = static_cast<uintmax_t>(static_cast<uintmax_t>(1U) << _coshift);
164 static constexpr uintmax_t _new_d = static_cast<uintmax_t>(D * _c1);
165 static constexpr uintmax_t _new_n0 = static_cast<uintmax_t>(N0 * _c1);
166 static constexpr uintmax_t _n1_shifted = static_cast<uintmax_t>((N1 % D) * _c1);
167 static constexpr uintmax_t _n0_top = static_cast<uintmax_t>((_shift != 0U) ? (N0 / _c2) : 0U);
168 static constexpr uintmax_t _new_n1 = static_cast<uintmax_t>(_n1_shifted + _n0_top);
169 using res_t = _big_div_impl<_new_n1, _new_n0, _new_d>;
170
171public:
172 static constexpr uintmax_t _quot_hi = N1 / D;
173 static constexpr uintmax_t _quot_lo = res_t::_quot;
174 static constexpr uintmax_t _rem = res_t::_rem / _c1;
175
176private:
177 using p0_t = _big_mul<_quot_lo, D>;
178 using p1_t = _big_mul<_quot_hi, D>;
