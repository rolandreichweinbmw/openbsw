457{

499{
500 if (contains(object))

502 size_type const pos = static_cast<size_type>(
503 std::distance(type_utils<T>::cast_const_to_type(_data), &object));

570template<class T>
571inline bool object_pool<T>::isUsed(size_type const pos) const
572{
573 size_type const i = static_cast<size_type>(pos / 8U);
574 size_type const shift = static_cast<size_type>(pos % 8U);
575 uint8_t const m = static_cast<uint8_t>(1U << shift);
576 return ((_used[i] & m) != 0U);
577}

582 size_type const i = static_cast<size_type>(pos / 8U);
583 size_type const shift = static_cast<size_type>(pos % 8U);
584 uint8_t const m = static_cast<uint8_t>(1U << shift);
585 _used[i] |= m;

588template<class T>
589inline void object_pool<T>::mark_unused(size_type const pos)
590{
591 size_type const i = static_cast<size_type>(pos / 8U);
592 size_type const shift = static_cast<size_type>(pos % 8U);
593 uint8_t const m = ~static_cast<uint8_t>(1U << shift);
594 _used[i] &= m;
595}

597template<class T>
598inline typename object_pool<T>::value_type*
599object_pool<T>::next(value_type const* const current) const
600{
601 if (current == nullptr)
602 {
603 return nullptr;
604 }
605 if (!contains(*current))
606 {
607 return nullptr;
608 }
609 size_type const offset = static_cast<size_type>(current - type_utils<T>::cast_to_type(_data));
610 for (size_type i = offset + 1; i < max_size(); ++i)
611 {
612 if (isUsed(i))
613 {
614 return type_utils<T>::cast_to_type(_data, i);
615 }
616 }
617 return nullptr;
618}
