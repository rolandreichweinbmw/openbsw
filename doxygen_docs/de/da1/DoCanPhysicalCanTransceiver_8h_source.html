131template<class Addressing>
132SendResult DoCanPhysicalCanTransceiver<Addressing>::startSendDataFrames(
133 FrameCodecType const& codec,
134 IDoCanDataFrameTransmitterCallback<DataLinkLayerType>& callback,
135 JobHandleType const jobHandle,
136 DataLinkAddressType const transmissionAddress,
137 FrameIndexType const firstFrameIndex,
138 FrameIndexType const lastFrameIndex,
139 FrameSizeType const consecutiveFrameDataSize,
140 ::estd::slice<uint8_t const> const& data)
141{
142 (void)lastFrameIndex;
143 if (!_sendPending)
144 {
145 ::estd::slice<uint8_t> payload = ::estd::slice<uint8_t>::from_pointer(
146 _frame.getPayload(), static_cast<size_t>(_frame.getMaxPayloadLength()));
147
148 if (codec.encodeDataFrame(
149 payload, data, firstFrameIndex, consecutiveFrameDataSize, _sendDataSize)
150 != CodecResult::OK)
151 {
152 return SendResult::INVALID;
153 }
154
155 uint32_t canId;
156 _addressing.encodeTransmissionAddress(transmissionAddress, canId, payload);
157 _frame.setId(canId);
158 _frame.setPayloadLength(static_cast<uint8_t>(payload.size()));
159 ::can::ICanTransceiver::ErrorCode const result = _transceiver.write(_frame, *this);
160 if (result == ::can::ICanTransceiver::ErrorCode::CAN_ERR_OK)
161 {
162 _sendPending = true;
163 _sendCallback = &callback;
164 _sendJobHandle = jobHandle;
165 return SendResult::QUEUED_FULL;
166 }
167 if (result != ::can::ICanTransceiver::ErrorCode::CAN_ERR_TX_HW_QUEUE_FULL)
168 {
169 return SendResult::FAILED;
170 }
171 // default return value: FULL
172 }
173 return SendResult::FULL;
174}

210{
211 if (_frameReceiver == nullptr)

216 ::estd::slice<uint8_t const> const payload = ::estd::slice<uint8_t const>::from_pointer(
217 canFrame.getPayload(), canFrame.getPayloadLength());
218 DataLinkAddressType const receptionAddress

220
221 DoCanTransportAddressPair transportAddressPair;
222 DataLinkAddressType transmissionAddress;
223 FrameCodecType const* const codec = _addressConverter.getReceptionParameters(
224 receptionAddress, transportAddressPair, transmissionAddress);
225 if (codec == nullptr)
226 {
227 return;
228 }
229
230 DoCanConnection<DataLinkLayerType> const connection(
231 *codec,
232 DoCanDataLinkAddressPair<DataLinkAddressType>(receptionAddress, transmissionAddress),
233 transportAddressPair);
234 (void)DoCanFrameDecoder<DataLinkLayerType>::decodeFrame(connection, payload, *_frameReceiver);
