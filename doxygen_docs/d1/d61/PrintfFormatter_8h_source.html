254template<class T>
255inline char* PrintfFormatter::formatIntDigits(
256 char* bufferEnd,
257 char const* const digits,
258 T value,
259 bool const signedType,
260 typename std::make_unsigned<T>::type const base,
261 int8_t& sign)
262{
263 if (value != 0U)
264 {
265 if (signedType)
266 {
267 if (static_cast<typename std::make_signed<T>::type>(value) < 0)
268 {
269 value = -value;
270 sign = -1;
271 }
272 else
273 {
274 sign = +1;
275 }
276 }
277 else
278 {
279 sign = +1;
280 }
281
282 while (value > 0U)
283 {
284 --bufferEnd;
285 *bufferEnd = digits[value % base];
286 value /= base;
287 }
288 }
289 else
290 {
291 sign = 0;
292 --bufferEnd;
293 *bufferEnd = '0';
294 }
295 return bufferEnd;
296}

299inline char const* PrintfFormatter::getIntSign(ParamInfo const& paramInfo, int8_t const sign)
300{
301 char const* pSign = "";
302 if (paramInfo._base == 10U)
303 {
304 switch (paramInfo._datatype)
305 {
306 case ParamDatatype::SINT8:
307 case ParamDatatype::SINT16:
308 case ParamDatatype::SINT32:
309 case ParamDatatype::SINT64:
310 {
311 if (sign < 0)
312 {
313 pSign = "-";
314 }
315 else if ((paramInfo._flags & ParamFlags::FLAG_PLUS) > 0U)
316 {
317 pSign = "+";
318 }
319 else if ((paramInfo._flags & ParamFlags::FLAG_SPACE) > 0U)
320 {
321 pSign = " ";
322 }
323 else
324 {
325 // do nothing
326 }
327 break;
328 }
329 default:
330 {
331 break;
332 }
333 }
334 }
335 return pSign;
336}

339inline char const* PrintfFormatter::getIntPrefix(ParamInfo const& paramInfo, int8_t const sign)
340{
341 char const* pPrefix = "";
342 switch (paramInfo._base)
343 {
344 case 8:
345 {
346 if ((sign > 0) && ((paramInfo._flags & ParamFlags::FLAG_ALT) != 0U))
347 {
348 pPrefix = "0";
349 }
350 break;
351 }
352 case 16:
353 {
354 if ((paramInfo._flags & ParamFlags::FLAG_ALT) != 0U)
355 {
356 pPrefix = ((paramInfo._flags & ParamFlags::FLAG_UPPER) > 0U) ? "0X" : "0x";
357 }
358 break;
359 }
360 default:
361 {
362 break;
363 }
364 }
365 return pPrefix;
366}
