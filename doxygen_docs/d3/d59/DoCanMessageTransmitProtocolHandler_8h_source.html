311template<typename FrameIndexType>
312inline TransmitResult DoCanMessageTransmitProtocolHandler<FrameIndexType>::frameSending()
313{
314 if (_state == TransmitState::SEND)
315 {
316 FrameIndexType const nextFrameIndex = _frameIndex + 1U;
317 if ((nextFrameIndex == _blockEnd) && (nextFrameIndex != _frameCount))
318 {
319 _flowControlWaitCount = 0U;
320 return setState(
321 TransmitState::WAIT, FlowControl::EXPECTED, TransmitTimeout::TX_CALLBACK);
322 }
323 return setState(TransmitState::WAIT, FlowControl::UNEXPECTED, TransmitTimeout::TX_CALLBACK);
324 }
325 return setFailed(TransmitMessage::ILLEGAL_STATE, static_cast<uint8_t>(_state));
326}

328template<typename FrameIndexType>
329inline TransmitResult
330DoCanMessageTransmitProtocolHandler<FrameIndexType>::framesSent(FrameIndexType const frameCount)
331{
332 if ((_state != TransmitState::WAIT) || (_timeout != TransmitTimeout::TX_CALLBACK))
333 {
334 return TransmitResult(true).setMessage(
335 TransmitMessage::ILLEGAL_STATE, static_cast<uint8_t>(_state));
336 }
337
338 _frameIndex += frameCount;
339 if (_frameIndex >= _frameCount)
340 {
341 return setState(TransmitState::SUCCESS);
342 }
343
344 if ((frameCount > 1U) && (_frameIndex >= _blockEnd))
345 {
346 _frameIndex = _blockEnd;
347 _flowControl = FlowControl::EXPECTED;
348 }
349 switch (_flowControl)
350 {
351 case FlowControl::RECEIVED_CTS:
352 {
353 return setSend();
354 }
355 case FlowControl::RECEIVED_WAIT:
356 case FlowControl::EXPECTED:
357 {
358 return setState(TransmitState::WAIT, _flowControl, TransmitTimeout::FLOW_CONTROL);
359 }
360 default:
361 {
362 if (_hasMinSeparationTime)
363 {
364 return setState(
365 TransmitState::WAIT, FlowControl::UNEXPECTED, TransmitTimeout::SEPARATION_TIME);
366 }
367 return setSend();
368 }
369 }

372template<typename FrameIndexType>
373inline TransmitResult DoCanMessageTransmitProtocolHandler<FrameIndexType>::handleFlowControl(
374 FlowStatus const flowStatus,
375 uint8_t const blockSize,
376 bool const hasMinSeparationTime,
377 uint8_t const maxWaitCount)
378{
379 if (_flowControl != FlowControl::EXPECTED)
380 {
381 return TransmitResult(false);
382 }
383
384 switch (flowStatus)
385 {
386 case FlowStatus::CTS:
387 {
388 _blockEnd = (blockSize > 0U) ? (_blockEnd + blockSize) : _frameCount;
389 _hasMinSeparationTime = hasMinSeparationTime;
390 if (_timeout == TransmitTimeout::FLOW_CONTROL)
391 {
392 (void)setSend(); // Result of setSend doesn't seem to matter, we know what to return
393 return TransmitResult(true).setActionSet(
394 TransmitActionSetType().set(TransmitAction::STORE_SEPARATION_TIME));
395 }
396
397 _flowControl = FlowControl::RECEIVED_CTS;
398 return TransmitResult(false).setActionSet(
399 TransmitActionSetType().set(TransmitAction::STORE_SEPARATION_TIME));
400 }
401 case FlowStatus::WAIT:
402 {
403 if (_flowControlWaitCount < maxWaitCount)
404 {
405 ++_flowControlWaitCount;
406 if (_timeout == TransmitTimeout::FLOW_CONTROL)
407 {
408 return setState(
409 TransmitState::WAIT, FlowControl::EXPECTED, TransmitTimeout::FLOW_CONTROL);
410 }
411
412 _flowControl = FlowControl::RECEIVED_WAIT;
413 return TransmitResult(false);
414 }
415
416 return setFailed(TransmitMessage::FLOW_CONTROL_WAIT_COUNT_EXCEEDED);
417 }
418 case FlowStatus::OVFLW:
419 {
420 return setFailed(TransmitMessage::FLOW_CONTROL_OVERFLOW);
421 }
422 default:
423 {
424 return setFailed(TransmitMessage::FLOW_CONTROL_INVALID);
425 }
426 }
427}
