241template<class Adapter>
242template<typename T>
243void TaskInitializer<Adapter>::create(
244 ContextType const context,
245 char const* const name,
246 TaskObjectType& task,
247 T& stack,
248 TaskFunctionType const taskFunction,
249 TaskConfigType const& config)
250{
251 ::estd::slice<uint8_t> bytes = ::estd::make_slice(stack).template reinterpret_as<uint8_t>();
252 ::estd::memory::align(alignof(StackType_t), bytes);
253 StackSliceType const stackSlice = bytes.template reinterpret_as<StackType_t>();
254 estd_assert((stackSlice.size() * sizeof(StackType_t)) >= sizeof(TaskInitializer));
255 new (stackSlice.data()) TaskInitializer(context, name, task, stackSlice, taskFunction, config);
256}
