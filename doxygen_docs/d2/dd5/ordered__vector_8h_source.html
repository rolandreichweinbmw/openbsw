516template<class T, class Comparator>
517typename ordered_vector<T, Comparator>::const_iterator
518ordered_vector<T, Comparator>::find_or_insert(const_reference value)
519{
520 const_iterator const position = std::lower_bound(cbegin(), cend(), value, _comp);
521 if ((position == cend()) || _comp(value, *position))
522 {
523 if (full())
524 {
525 return end();
526 }
527 (void)_values.insert(position, value);
528 }
529 return position;
530}

554template<class Predicate>
555void ordered_vector<T, Comparator>::remove_if(Predicate pred)
556{
557 const_iterator b = cbegin();
558 // This is a safe way to remove elements from a container while iterating over the container
559
560 while (b != cend())
561 {
562 if (pred(*b) == true)
563 {
564 b = _values.erase(b);
565 }
566 else
567 {
568 ++b;
569 }
570 }
