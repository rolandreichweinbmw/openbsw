42template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE = uint16_t>
43class MemoryQueue
44{
45 static_assert(CAPACITY >= MAX_ELEMENT_SIZE + sizeof(SIZE_TYPE), "");
46
47 struct RxData
48 {
49 std::atomic<size_t> received{0U};
50 };
51
52 RxData rx;
53
54 struct TxData
55 {
56 std::atomic<size_t> sent{0U};
57 ::estd::array<uint8_t, CAPACITY> data;
58 size_t allocated{0U};
59 size_t minAvailable{CAPACITY};
60
61 size_t available(RxData const& rxData) const;
62 };
63
64 TxData tx;

66 static size_t advanceIndex(size_t index, size_t const size)
67 {
68 index = (index + size + sizeof(SIZE_TYPE)) % (2 * CAPACITY);
69
70 // Check if enough contiguous space is available at the end of the array and wrap
71 // otherwise skipping the too small piece of memory.
72 size_t const space = (((2 * CAPACITY) - index) % CAPACITY);
73 if (space < (MAX_ELEMENT_SIZE + sizeof(SIZE_TYPE)))
74 {
75 index = (index + space) % (2 * CAPACITY);
76 }
77 return index;
78 }

233template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
234::estd::slice<uint8_t>
235MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::Writer::allocate(size_t const size) const
236{
237 // Set allocated zero prevent a subsequent call to commit() to have
238 // effects in case this allocation fails. That is important if this is a
239 // reallocation, i.e. a previous call to allocate() succeeded.
240 _txData.allocated = 0U;
241 if ((size > MAX_ELEMENT_SIZE) || (size == 0))
242 {
243 return {};
244 }
245 auto const freeBytes = available();
246 _txData.minAvailable = ::estd::min(freeBytes, _txData.minAvailable);
247 if (freeBytes == 0)
248 {
249 return {};
250 }
251 size_t const index = _txData.sent.load() % CAPACITY;
252 _txData.allocated = size;
253 return ::estd::memory::as_bytes(&_txData.data[index + sizeof(SIZE_TYPE)], size);
254}

256template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
257void MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::Writer::commit()
258{
259 // Prevent accidentally committing random data if allocate has not been called or
260 // previous allocation was unsuccessful.
261 if (_txData.allocated == 0U)
262 {
263 return;
264 }
265 size_t writeIndex = _txData.sent.load();
266 size_t const index = writeIndex % CAPACITY;
267 SIZE_TYPE const size = static_cast<SIZE_TYPE>(_txData.allocated);
268 ::estd::write_be<SIZE_TYPE>(&_txData.data[index], static_cast<SIZE_TYPE>(size));
269
270 writeIndex = advanceIndex(writeIndex, size);
271 _txData.allocated = 0U;
272 // Store writeIndex last to ensure data consistency.
273 _txData.sent.store(writeIndex);
274}

332template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
333void MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::Reader::release() const
334{
335 if (empty())
336 {
337 return;
338 }
339 size_t readIndex = _rxData.received.load();
340 size_t const index = readIndex % CAPACITY;
341 SIZE_TYPE const size = ::estd::read_be<SIZE_TYPE>(&_txData.data[index]);
342 readIndex = advanceIndex(readIndex, size);
343 // Store readIndex last to ensure data consistency.
344 _rxData.received.store(readIndex);
345}

368template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
369size_t
370MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::TxData::available(RxData const& rxData) const
371{
372 size_t const writeIndex = sent.load();
373 size_t const readIndex = rxData.received.load();
374
375 size_t usedBytes;
376 if (writeIndex < readIndex)
377 {
378 usedBytes = (writeIndex + (2 * CAPACITY)) - readIndex;
379 }
380 else
381 {
382 usedBytes = writeIndex - readIndex;
383 }
384
385 size_t freeBytes = CAPACITY - usedBytes;
386 if (freeBytes < (MAX_ELEMENT_SIZE + sizeof(SIZE_TYPE)))
387 {
388 freeBytes = 0;
389 }
390 return freeBytes;
391}
