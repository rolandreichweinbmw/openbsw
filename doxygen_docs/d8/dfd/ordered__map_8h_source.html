481template<class Key, class T, class Compare>
482inline bool
483operator==(ordered_map<Key, T, Compare> const& lhs, ordered_map<Key, T, Compare> const& rhs)
484{
485 if (lhs.size() != rhs.size())
486 {
487 return false;
488 }
489 for (typename ordered_map<Key, T, Compare>::const_iterator itr = lhs.cbegin();
490 itr != lhs.cend();
491 ++itr)
492 {
493 if (rhs.count(itr->first) > 0U)
494 {
495 if (!(rhs.at(itr->first) == itr->second))
496 {
497 return false;
498 }
499 }
500 else
501 {
502 return false;
503 }
504 }
505 return true;
506}

598template<class Key, class T, class Compare>
599std::pair<typename ordered_map<Key, T, Compare>::iterator, bool>
600ordered_map<Key, T, Compare>::insert(const_reference value)
601{
602 iterator const pos = lower_bound(value.first);
603 if ((pos != end()) && key_eq(pos->first, value.first))
604 {
605 return std::make_pair(pos, false);

667template<class Key, class T, class Compare>
668typename ordered_map<Key, T, Compare>::iterator
669ordered_map<Key, T, Compare>::erase(const_iterator const first, const_iterator const last)
670{
671 estd_assert(last <= cend());
672 if (last <= first)
673 {
674 return begin() + (last - cbegin()); // empty range
675 }
676
677 // copy elements after range to front of range
678 iterator const returnItem = begin() + (first - cbegin());
679 iterator dstIter = returnItem;
680 iterator srcIter = begin() + (last - cbegin());
681 while (srcIter != end())
682 {
683 dstIter->~value_type();
684 (void)constructor<value_type>(type_utils<value_type>::cast_to_raw(dstIter))
685 .construct(::std::move(*srcIter));
686 ++dstIter;
687 ++srcIter;
688 }
689
690 // destroy excess tail elements
691 auto const numToPop = static_cast<size_type>(static_cast<difference_type>(last - first));
692 for (size_type i = 0; i < numToPop; ++i)
693 {
694 _values.pop_back();
695 }
696 return returnItem;
697}

939template<class Key, class T, class Compare>
940typename ordered_map<Key, T, Compare>::size_type
941ordered_map<Key, T, Compare>::lower_bound_search(key_type const& key) const
942{
943 size_type pos;
944 size_type first = 0U;
945 difference_type total = static_cast<difference_type>(_values.size());
946 difference_type step;
947
948 while (total > 0)
949 {
950 pos = first;
951 step = total / 2;
952 pos += static_cast<size_type>(step);
953 if (_comp(_values[pos].first, key) == true)
954 {
955 pos++;
956 first = pos;
957 total -= step + 1;
958 }
959 else
960 {
961 total = step;
962 }
963 }
964 return first;
965}

967template<class Key, class T, class Compare>
968typename ordered_map<Key, T, Compare>::size_type
969ordered_map<Key, T, Compare>::upper_bound_search(key_type const& key) const
970{
971 size_type pos;
972 size_type first = 0U;
973 difference_type total = static_cast<difference_type>(_values.size());
974 difference_type step;
975
976 while (total > 0)
977 {
978 pos = first;
979 step = total / 2;
980 pos += static_cast<size_type>(step);
981 if (_comp(key, _values[pos].first) == false)
982 {
983 pos++;
984 first = pos;
985 total -= step + 1;
986 }
987 else
988 {
989 total = step;
990 }
991 }
992 return first;
993}

995template<class Key, class T, class Compare>
996inline typename ordered_map<Key, T, Compare>::value_type&
997ordered_map<Key, T, Compare>::emplace_uninitialized(ordered_map::iterator const pos)
998{
999 estd_assert(!full());
1000 if (pos == cend())
1001 {
1002 return *(_values.emplace_back().release_uninitialized_as());
1003 }
1004
1005 // move all elements after pos to the back by one
1006 auto dstItr = end() - 1;
1007 auto srcItr = dstItr - 1;
1008 _values.emplace_back().construct(::std::move(*dstItr)); // construct new last element
1009 dstItr->~value_type(); // destroy previous last element
1010 while (dstItr != pos)
1011 {
1012 // move src to dest and destroy src
1013 (void)constructor<ordered_map::value_type>(
1014 type_utils<ordered_map::value_type>::cast_to_raw(dstItr))
1015 .construct(::std::move(*srcItr));
1016 srcItr->~value_type();
1017 --dstItr;
1018 --srcItr;
1019 }
1020 return *pos;
1021}
