295template<class Binding>
296void FreeRtosAdapter<Binding>::initTask(TaskInitializer& initializer)
297{
298 ContextType const context = initializer._context;
299 _stackSizes[static_cast<size_t>(context)] = static_cast<uint32_t>(
300 static_cast<size_t>(initializer._stack.size()) * sizeof(BaseType_t));
301 TaskConfigsType::setTaskConfig(context, initializer._config);
302 if (context == TASK_TIMER)
303 {
304 _timerTaskInitializer = &initializer;
305 _timerTaskName = initializer._name;
306 }
307 else
308 {
309 TaskContextType& taskContext = _taskContexts[static_cast<size_t>(context)];
310 if (context == TASK_IDLE)
311 {
312 _idleTaskInitializer = &initializer;
313 taskContext.initTask(TASK_IDLE, initializer._name, initializer._taskFunction);
314 }
315 else
316 {
317 taskContext.createTask(
318 context,
319 initializer._task,
320 initializer._name,
321 static_cast<UBaseType_t>(context),
322 initializer._stack,
323 initializer._taskFunction);
324 }
325 }

346template<class Binding>
347bool FreeRtosAdapter<Binding>::getStackUsage(size_t const taskIdx, StackUsage& stackUsage)
348{
349 if (taskIdx < FREERTOS_TASK_COUNT)
350 {
351 stackUsage._stackSize = _stackSizes[taskIdx];
352 uint32_t const unusedSize
353 = (taskIdx == TASK_TIMER)
354 ? TaskContextType::getUnusedStackSize(xTimerGetTimerDaemonTaskHandle())
355 : _taskContexts[taskIdx].getUnusedStackSize();
356 stackUsage._usedSize = stackUsage._stackSize - unusedSize;
357 return true;
358 }
359 return false;
360}

422template<class Binding>
423inline void FreeRtosAdapter<Binding>::leaveIsr()
424{
425 internal::NestedInterruptLock<> const lock;
426 --_nestedInterruptCount;
427 if (_nestedInterruptCount == 0U)
428 {
429 FreeRtosAdapter::_higherPriorityTaskWoken = nullptr;
430 if (FreeRtosAdapter::_higherPriorityTaskWokenFlag != pdFALSE)
431 {
432 FreeRtosAdapter::_higherPriorityTaskWokenFlag = pdFALSE;
433 portYIELD_FROM_ISR(pdTRUE);
434 }
435 }
436}
