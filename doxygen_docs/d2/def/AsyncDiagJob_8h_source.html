87template<class T>
88DiagReturnCode::Type AsyncDiagJob<T>::process(
89 IncomingDiagConnection& connection, uint8_t const request[], uint16_t const requestLength)
90{
91 if (fAsyncJobHelper.hasPendingAsyncRequest())
92 {
93 return fAsyncJobHelper.enqueueRequest(connection, request, requestLength);
94 }
95 fAsyncJobHelper.startAsyncRequest(connection);
96 fProcess = ProcessClosureType::CallType(
97 ProcessClosureType::CallType::fct::create<AsyncDiagJob<T>, &AsyncDiagJob<T>::asyncProcess>(
98 *this),
99 &connection,
100 request,
101 requestLength);
102
103 ::async::execute(fContext, fProcess);
104
105 return DiagReturnCode::OK;
106}

108template<class T>
109void AsyncDiagJob<T>::asyncProcess(
110 IncomingDiagConnection* connection, uint8_t const request[], uint16_t const requestLength)
111{
112 DiagReturnCode::Type responseCode = T::process(*connection, request, requestLength);
113 if (responseCode != DiagReturnCode::OK)
114 {
115 connection->sendNegativeResponse(responseCode, *this);
116 connection->terminate();
117 }
