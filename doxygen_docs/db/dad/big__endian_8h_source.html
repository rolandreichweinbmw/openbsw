185template<typename T, uint32_t N = sizeof(T)>
186struct big_endian
187{
188 static_assert(N <= sizeof(T), "");
189
190 uint8_t bytes[N];
191
192 big_endian& operator=(T const value)
193 {
194 internal::write_be<T, N>(&bytes[0], value);
195 return *this;
196 }
197
198 operator T() const { return internal::read_be<T, N>(&bytes[0]); }
199
200 static big_endian make(T const value)
201 {
202 big_endian ret;
203 ret = value;
204 return ret;
205 }
206};

239template<typename T>
240T read_be_bits(uint8_t const* const src, size_t offset, size_t length)
241{
242 estd_assert(length <= sizeof(T) * 8);
243 T result = 0;
244 // Adjust offset to modulo of 8 for ease of calculation, and set index accordingly
245 size_t idx = offset / 8;
246 offset = offset % 8;
247 // Shift value for mask: as it's potentially only part of the first byte, min between length and
248 // Number of bytes in first byte to read
249 uint8_t const firstLength = static_cast<uint8_t>(::estd::min<size_t>(length, 8U - offset));
250 // Shift value for return value: 8 - (offset + length), or 0 if offset + length >= 8
251 uint8_t const firstShift = 8U - static_cast<uint8_t>(::estd::min<size_t>(8U, offset + length));
252 // Take uint8_t max (all 1s), shift by number of bits to read, then invert (all 0s for length 0)
253 uint8_t const firstMask = ~(::std::numeric_limits<uint8_t>::max() << firstLength);
254
255 // Set result to first byte, shifted & masked
256 result = static_cast<T>((src[idx] >> firstShift) & firstMask);
257 length -= firstLength;
258 ++idx;
259
260 while (length != 0)
261 {
262 if (length >= 8)
263 {
264 // Full byte copy, no shifting or masking needed
265 if (sizeof(T) > 1)
266 {
267 result <<= 8;
268 }
269 result |= static_cast<T>(src[idx]);
270 ++idx;
271 length -= 8;
272 }
273 else
274 {
275 // Final partial byte read
276 uint8_t const lastShift = static_cast<uint8_t>(8 - length);
277 uint8_t const lastMask = std::numeric_limits<uint8_t>::max() >> lastShift;
278 result <<= length;
279 result |= static_cast<T>((src[idx] >> lastShift) & lastMask);
280 ++idx;
281 // Consume the rest of the length here. Same as length -= length
282 length = 0;
283 }
284 }
285 return result;
286}

294template<typename T>
295void write_be_bits(uint8_t* const dst, T src, size_t offset, size_t length)
296{
297 estd_assert(length <= sizeof(T) * 8);
298 // Adjust offset to modulo of 8 for ease of calculation, and set index accordingly
299 size_t idx = (offset + length) / 8;
300 offset = offset % 8;
301
302 // Start with last byte written. This code will only run when a write spans multiple bytes
303 if (length >= (8 - offset))
304 {
305 // Calculate the number of bits the final byte read will use
306 uint8_t const endLength = static_cast<uint8_t>((length + offset - 8) % 8);
307 if (endLength != 0)
308 {
309 // set lowest bits to 1
310 uint8_t const srcEndMask = ::std::numeric_limits<uint8_t>::max() >> (8 - endLength);
311 // set highest bits to 1
312 uint8_t const dstEndMask = ::std::numeric_limits<uint8_t>::max() << (8 - endLength);
313
314 // Clear the bits we'll be setting, in case there's anything non-zero there
315 dst[idx] &= ~dstEndMask;
316 // Mask src, then shift over to correct position and bitwise or into dest
317 dst[idx] |= ((static_cast<uint8_t>(src) & srcEndMask) << (8 - endLength));
318 src >>= endLength;
319 length -= endLength;
320 }
321 --idx;
322 }
323
324 while (length != 0)
325 {
326 if (length >= 8)
327 {
328 // Write a full byte at a time
329 dst[idx] = static_cast<uint8_t>(src) & 0xFFU;
330 if (sizeof(T) > 1)
331 {
332 src >>= 8;
333 }
334 --idx;
335 length -= 8;
336 }
337 else
338 {
339 // Write the first byte
340 // length < 8 guaranteed (by above conditional)
341 uint8_t dstStartMask = ::std::numeric_limits<uint8_t>::max() >> length;
342 // Flip dstStartMask to set highest bits, shift by offset to shift mask into correct
343 // starting position
344 dstStartMask = static_cast<uint8_t>(~dstStartMask) >> offset;
345 // srcStartMask is just a mask for the lowest "length" # of bits in src
346 uint8_t const srcStartMask = ::std::numeric_limits<uint8_t>::max() >> (8 - length);
347 uint8_t const dstStartShift = 8U - static_cast<uint8_t>(offset + length);
348
349 // Clear lowest starting bits in case they contain anything non-zero
350 dst[idx] &= ~dstStartMask;
351 // Mask, shift, and or the src into the dst
352 dst[idx] |= (static_cast<uint8_t>(src) & srcStartMask) << dstStartShift;
353 // Same as length -= length
354 length = 0;
355 }
356 }
357}
