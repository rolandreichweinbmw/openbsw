85inline ptrdiff_t compare(slice<uint8_t const> const a, slice<uint8_t const> const b)
86{
87 auto const minSize = min(a.size(), b.size());
88 if (minSize > 0)
89 {
90 auto const pointerDiff = static_cast<ptrdiff_t>(::memcmp(a.data(), b.data(), minSize));
91
92 if (pointerDiff != 0)
93 {
94 return pointerDiff;
95 }
96 }
97
98 return static_cast<ptrdiff_t>(b.size()) - static_cast<ptrdiff_t>(a.size());
99}

101inline void reverse_bytes(slice<uint8_t> const buf)
102{
103 slice<uint8_t>::iterator lower = buf.begin();
104 slice<uint8_t>::iterator upper = buf.end() - 1;
105 uint8_t swap;
106
107 while (lower < upper)
108 {
109 swap = *lower;
110 *lower++ = *upper;
111 *upper-- = swap;
112 }
113}
