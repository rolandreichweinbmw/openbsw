297template<class Binding>
298inline EventMaskType TaskContext<Binding>::waitEvents()
299{
300 EventMaskType eventMask = 0U;
301 uint32_t ticks = Binding::WAIT_EVENTS_TICK_COUNT;
302 uint32_t nextDelta;
303 bool const hasDelta = _timer.getNextDelta(getSystemTimeUs32Bit(), nextDelta);
304 if (hasDelta)
305 {
306 ticks = static_cast<uint32_t>((nextDelta + (Config::TICK_IN_US - 1U)) / Config::TICK_IN_US);
307 }
308 if (xTaskNotifyWait(0U, WAIT_EVENT_MASK, &eventMask, ticks) != 0)
309 {
310 return eventMask;
311 }
312 else if (hasDelta)
313 {
314 return TimerEventPolicyType::EVENT_MASK;
315 }
316 else
317 {
318 return 0U;
319 }
320}
